		GetTypename();
	}
	void Initialize(Base* pointer) {
		GetTypename(pointer);
		pointer->display_not_virtual();
		(*pointer).display_not_virtual();
	}
	string type_name;
	void GetTypename() {
		type_name = typeid(this).name();
	}
	void GetTypename(Base* pointer) {
		type_name = typeid(*pointer).name();
	}
	virtual void display() {
		cout << "This is Base Class" << endl;
	}

		cout << "This is Base Class Not virtual" << endl;
	}
};
